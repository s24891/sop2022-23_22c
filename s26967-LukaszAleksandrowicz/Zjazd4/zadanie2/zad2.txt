#include <iostream>
#include <fstream>
#include <vector>

int main(int argc, char* argv[]) {
    std::vector<int> nums;
    std::istream* in;

    if (argc > 1) {
        std::ifstream file(argv[1]);
        if (!file.is_open()) {
            std::cerr << "Cannot open file " << argv[1] << std::endl;
            return 1;
        }
        in = &file;
    } else {
        in = &std::cin;
    }

    int n;
    *in >> n;

    int x;
    for (int i = 0; i < n; i++) {
        *in >> x;
        nums.push_back(x);
    }

    for (int i = n - 1; i >= 0; i--) {
        std::cout << nums[i] << std::endl;
    }

    return 0;
}
