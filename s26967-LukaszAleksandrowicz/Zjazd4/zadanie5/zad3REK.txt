#include <iostream>

int fib(int n) {
    if (n <= 1) {
        return n;
    }
    return fib(n-1) + fib(n-2);
}

int main() {
    int n = 10;
    std::cout << "Ciag fibonacciego do 10 liczby: ";
    for (int i = 0; i < n; i++) {
        std::cout << fib(i) << " ";
    }
    std::cout << std::endl;
    return 0;
}
