#include <iostream>

int main() {
    int n, sum = 0;
    std::cout << "Podaj ilość liczb: ";
    std::cin >> n;
    for (int i = 0; i < n; i++) {
        int x;
        std::cout << "Podaj liczbę " << i+1 << ": ";
        std::cin >> x;
        sum += x;
    }
    std::cout << "Suma liczb wynosi: " << sum << std::endl;
    return sum;
}
